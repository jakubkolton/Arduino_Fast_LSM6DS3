59 int LSM6DS3 ::getRegister(uint8_t addr, uint8_t &data)
60 {
61  I2C->beginTransmission(I2C_Address); // rozpoczecie transmisji w kierunku Slave - wyslanie jego adresu + bitu Read
62  I2C->write(addr); // wyslanie adresu rejestru
63 
64  // Obsluga bledu transmisji
65  if (0 != I2C->endTransmission(false))
66  {
67  return -1; // kod bledu
68  }
69 
70  // Obsluga zbyt malej liczby odebranych bajtow
71  if (1 != I2C->requestFrom(I2C_Address, 1))
72  {
73  return 0; // kod bledu
74  }
75 
76  // Odczyty rejestru - jednego bajtu
77  data = I2C->read();
78 
79  return 1; // kod poprawnego wykonania
80 }

112 // Metoda odczytu pomiaru z akcelerometru (dla dowolnych parametrow)
113 int LSM6DS3 ::readAcceleration(float &x, float &y, float &z)
114 {
115  int16_t dataReg; // bufor na odczytywany rejestr
116  uint8_t MSB, LSB; // bufor na MSB i LSB
117  uint8_t CTRL1_XL; // bufor na rejestr CTRL1_XL
118  uint8_t range; // zakres pomiarowy
119 
120  // Zakres pomiarowy
121  if (1 != getRegister(LSM6DS3_CTRL1_XL, CTRL1_XL))
122  return -1; // kod bledu
123  else
124  {
125  range = (CTRL1_XL & 0x0C) >> 2;
126  switch (range)
127  {
128  case LSM6DS3_FS_XL_2:
129  range = 2;
130  break;
131  case LSM6DS3_FS_XL_4:
132  range = 4;
133  break;
134  case LSM6DS3_FS_XL_8:
135  range = 8;
136  break;
137  case LSM6DS3_FS_XL_16:
138  range = 16;
139  break;
140  default:
141  break;
142  }
143  }
144 
145  // Oś X
146  if ((1 != getRegister(LSM6DS3_OUTX_L_XL, LSB)) ||
147  (1 != getRegister(LSM6DS3_OUTX_H_XL, MSB)))
148  x = NAN; // blad - not a number
149  else
150  {
151  dataReg = (MSB << 8) | LSB;
152  x = dataReg * range / 32768.0; // ze wzoru na przyspieszenie
153  }
154 
155  // Oś Y
156  if ((1 != getRegister(LSM6DS3_OUTY_L_XL, LSB)) ||
157  (1 != getRegister(LSM6DS3_OUTY_H_XL, MSB)))
158  y = NAN; // blad - not a number
159  else
160  {
161  dataReg = (MSB << 8) | LSB;
162  y = dataReg * range / 32768.0; // ze wzoru na przyspieszenie
163  }
164 
165  // Oś X
166  if ((1 != getRegister(LSM6DS3_OUTZ_L_XL, LSB)) ||
167  (1 != getRegister(LSM6DS3_OUTZ_H_XL, MSB)))
168  z = NAN; // blad - not a number
169  else
170  {
171  dataReg = (MSB << 8) | LSB;
172  z = dataReg * range / 32768.0; // ze wzoru na przyspieszenie
173  }
174 
175  return 1;
176 }
